def reverseList(self, head: ListNode) -> ListNode:
    # 定义一个前置节点
    pre = None
    # cur指向当前头指针
    cur = head
    while cur != None:
        # 每次都用temp来存链表的下一个
        temp = cur.next
        # cur指向pre
        cur.next = pre
        # 移动 per、cur
        pre = cur
        cur = temp
    return pre
