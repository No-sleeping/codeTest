**这里主要采用双指针方法，来不断刷新数组（覆盖旧数据）。**

双指针主要用法：

- **双头指针** 
  ```python
  fast, slow = 0, 0
  ```
- **首尾指针**
  ```python
  left, right = 0, len(nums)-1
  ```

```python
# 27 移除元素
def removeElement(nums,val):
    n = len(nums)
    fast, slow = 0, 0
    while fast < n:
        if nums[fast] != val:
            # 若不是待删除元素，快指针给慢指针赋值
            nums[slow] = nums[fast]
            slow += 1
            #若是删除元素，慢指针不动，快指针继续寻找下一个非删除元素赋值过来
        fast += 1
    return slow
```

```python
# 26.删除排序数组中的重复项
def removeDuplicates(nums):
    n = len(nums)
    fast, slow = 1, 0
    while fast < n :
        if nums[fast] == nums[slow]:
            # 相等的时候，快指针加一
            fast += 1
        else:
            # 不相等时，先慢指针加一，快给慢指针赋值，快指针加一
            slow += 1
            nums[slow] = nums[fast]
            fast += 1
            # 这里slow是最后一个不重复元素的下标，要求返回的是长度。
            # 所以这里返回要加一
    return slow+1
```

```python
# 283 移动零
def moveZeroes(nums):
    n = len(nums)
    fast, slow = 0, 0
    # 第一遍刷双指针把非0元素都移动到靠前位置。
    for fast in range(n):
        if nums[fast] == 0:
            fast += 1
        else:
            slow += 1
    # 第二遍历，把slow 后面的元素全部置零，也就是把非0元素以外的元素置零。
    for i in range(slow,n):
        nums[i] = 0
    return nums
```

```python
#  844 - 比较含退格的字符串
def backspaceCompare(nums1,nums2):
    def judge(nums):
        n = len(nums)
        slow, fast = 0, 0
        for fast in range(n):
            if nums[fast] != "#":
                # 如果快指针遇到非“#”，不需要回退
                # 把当前快指针元素替换给慢指针对应元素
                # 这里注意只replace一个元素！
                nums = nums.replace(nums[slow],nums[fast],1)
                slow += 1
            else:
                # 如果快指针遇到“#”，回退
                slow -= 1
                # 这里判断slow < 0, 返回空字符串
        return nums[:slow] if slow > 0 else ""
    return judge(nums1) == judge(nums2)
```
